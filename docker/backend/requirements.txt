# docker/requirements.txt
# Liste des dépendances Python pour B'Craft'D Backend
# Pensez à mettre à jour ce fichier et rebuild l'image après modifications

# ============================================
# FRAMEWORK WEB - FastAPI
# ============================================
# FastAPI : framework web moderne et performant pour créer des APIs REST
# Inclut la validation automatique avec Pydantic
fastapi[all]==0.122.0

# Uvicorn : serveur ASGI haute performance pour exécuter FastAPI
# standard : inclut les dépendances pour le hot-reload et les websockets
uvicorn[standard]==0.38.0

# ============================================
# BASE DE DONNÉES - PostgreSQL
# ============================================
# SQLAlchemy : ORM (Object Relational Mapping) pour interagir avec PostgreSQL
# Permet de manipuler la base de données avec des objets Python
sqlalchemy==2.0.36

# Psycopg2-binary : driver PostgreSQL pour Python
# Version binaire (plus facile à installer que psycopg2)
psycopg2-binary==2.9.10

# Alembic : outil de migration de base de données
# Gère les versions du schéma de la base de données
alembic==1.17.2

# ============================================
# SÉCURITÉ - Authentification & JWT
# ============================================
# Passlib : bibliothèque de hachage de mots de passe
# Supporte bcrypt, argon2, scrypt, etc.
passlib==1.7.4

# Bcrypt : hachage sécurisé des mots de passe
# Version compatible avec la dernière version de passlib
bcrypt==4.1.2

# Python-jose : bibliothèque JWT (JSON Web Tokens)
# Utilisée pour générer et vérifier les tokens d'authentification
python-jose[cryptography]==3.3.0

# ============================================
# UTILITAIRES
# ============================================
# Python-dotenv : chargement automatique des variables d'environnement depuis fichier .env
python-dotenv==1.2.0
